# 1. Build the Docker image
docker build -t hwco:latest .

# 2. run container with persistent data volume
docker run -d \
  --name hwco \
  -v /Users/fijsheijkoop/orbstack-volumes/hwco-data:/data \
hwco:latest

# 3. ensure volume directories exist and have correct permissions  
mkdir -p "$HOME/orbstack-volumes/hwco-data/config" "$HOME/orbstack-volumes/hwco-data/logs"
chmod -R u+rwX "$HOME/orbstack-volumes/hwco-data"
   
# run hwco reporter
docker exec -it hwco dotnet hwco.dll -r
